select
c.customer_id,
c.customer_name
from Customers c
LEFT JOIN orders o on c.customer_id = o.customer_id
group by c.customer_id
HAVING SUM(product_name='A') > 0
    AND SUM(product_name='B') > 0
    AND SUM(product_name='C') = 0
ORDER BY c.customer_id;
